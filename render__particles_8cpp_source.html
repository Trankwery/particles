116 void ParticleRenderer::_drawPoints(int pFrom, int pTo)
117 {
118  if (!m_vbo)/**< zarejestrowanie miejsc w pamięci gdzie są współżędne punktów */
119  {
120  glBegin(GL_POINTS);
121  {
122  int k = pFrom*4;
123 
124  for (int i = pFrom; i < pTo; ++i)
125  {
126  glVertex3fv(&m_pos[k]);
127  k += 4;
128  }
129  }
130  glEnd();
131  }
132  else
133  {
134  glBindBufferARB(GL_ARRAY_BUFFER_ARB, m_vbo);
135  glVertexPointer(4, GL_FLOAT, 0, 0);
136  glEnableClientState(GL_VERTEX_ARRAY);
137 
138  if (m_colorVBO && multiColor)/**< nie używamy bufora kolorów jeśli wszystkie cząstki z grupy mają być w jednym kolorze. */
139  {
140  glBindBufferARB(GL_ARRAY_BUFFER_ARB, m_colorVBO);
141  glColorPointer(4, GL_FLOAT, 0, 0);
142  glEnableClientState(GL_COLOR_ARRAY);
143  }
144 
145  glDrawArrays(GL_POINTS, pFrom, (pTo-pFrom));/**< rysowanie fragmentu bufora */
146 
147  glBindBufferARB(GL_ARRAY_BUFFER_ARB, 0);
148  glDisableClientState(GL_VERTEX_ARRAY);
149  glDisableClientState(GL_COLOR_ARRAY);
150  }
151 }

159 void ParticleRenderer::display(DisplayMode mode /* = PARTICLE_POINTS */)
160 {
161 /*#ifdef _DEBUG
162  system("pause");
163  std::clog<<"typyCzastek.size() "<<typyCzastek.size()<<"\n";
164 #endif*/
165  int nextType=0;
166  int typeBegin=0;
167  switch (mode)
168  {
169  case PARTICLE_POINTS:
170  glColor3f(1, 1, 1);
171  glPointSize(m_pointSize);
172  _drawPoints();

174 
175  default:
176  case PARTICLE_SPHERES:
177  glEnable(GL_POINT_SPRITE_ARB);
178  glTexEnvi(GL_POINT_SPRITE_ARB, GL_COORD_REPLACE_ARB, GL_TRUE);
179  glEnable(GL_VERTEX_PROGRAM_POINT_SIZE_NV);
180  glDepthMask(GL_TRUE);
181  glEnable(GL_DEPTH_TEST);
182 
183  glUseProgram(m_program);
184  for(int i=0;i<typyCzastek.size();i++)/**< rysujemy cząstki grupami. Wszystkie cząstki z jednego typu za raz ze wspólnymi cechami (promień i kolor) */
185  {
186 
187  glUniform1f(glGetUniformLocation(m_program, "pointScale"), zoom*m_window_h / tanf(m_fov*0.5f*(float)M_PI/180.0f));/**< skalowanie rozmiaru cząstki do ekranu */
188  glUniform1f(glGetUniformLocation(m_program, "pointRadius"), typyCzastek[i].particleRadius);/**< nadanie grupie cząstek jednego promienia */
189 
190  glColor3f(typyCzastek[i].particleColorR, typyCzastek[i].particleColorG, typyCzastek[i].particleColorB);/**< ustalenie jednego koloru dla grupy cząstek */
191 
192  nextType+=typyCzastek[i].particleNoOfType;
193  _drawPoints(typeBegin, nextType);/**< wywołanie rysowania grupy cząstek */
194  typeBegin=nextType;
195 
196  }
197  glUseProgram(0);
198 
199  glDisable(GL_POINT_SPRITE_ARB);
200  break;
201  }

205 ParticleRenderer::_compileProgram(const char *vsource, const char *fsource)
206 {
207  GLuint vertexShader = glCreateShader(GL_VERTEX_SHADER);
208  GLuint fragmentShader = glCreateShader(GL_FRAGMENT_SHADER);
209 
210  glShaderSource(vertexShader, 1, &vsource, 0);
211  glShaderSource(fragmentShader, 1, &fsource, 0);
212 
213  glCompileShader(vertexShader);
214  glCompileShader(fragmentShader);
215 
216  GLuint program = glCreateProgram();
217 
218  glAttachShader(program, vertexShader);
219  glAttachShader(program, fragmentShader);
220 
221  glLinkProgram(program);
222 
223  // check if program linked
224  GLint success = 0;
225  glGetProgramiv(program, GL_LINK_STATUS, &success);
226 
227  if (!success)
228  {
229  char temp[256];
230  glGetProgramInfoLog(program, 256, 0, temp);
231  printf("Failed to link program:\n%s\n", temp);
232  glDeleteProgram(program);
233  program = 0;
234  }
235 
236  return program;
237 }
