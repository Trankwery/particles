194 void runBenchmark(int iterations, char *exec_path)
195 {
196  printf("Run %u particles simulation for %d iterations...\n\n", numParticles, iterations);
197  cudaDeviceSynchronize();
198  sdkStartTimer(&timer);
199 
200  for (int i = 0; i < iterations; ++i)
201  {
202  psystem->update(timestep);
203  }
204 
205  cudaDeviceSynchronize();
206  sdkStopTimer(&timer);
207  float fAvgSeconds = ((float)1.0e-3 * (float)sdkGetTimerValue(&timer)/(float)iterations);
208 
209  printf("particles, Throughput = %.4f KParticles/s, Time = %.5f s, Size = %u particles, NumDevsUsed = %u, Workgroup = %u\n",
210  (1.0e-3 * numParticles)/fAvgSeconds, fAvgSeconds, numParticles, 1, 0);
211 
212  if (g_refFile)
213  {
214  printf("\nChecking result...\n\n");
215  float *hPos = (float *)malloc(sizeof(float)*4*psystem->getNumParticles());
216  copyArrayFromDevice(hPos, psystem->getCudaPosVBO(),
217  0, sizeof(float)*4*psystem->getNumParticles());
218 
219  sdkDumpBin((void *)hPos, sizeof(float)*4*psystem->getNumParticles(), "particles.bin");
220 
221  if (!sdkCompareBin2BinFloat("particles.bin", g_refFile, sizeof(float)*4*psystem->getNumParticles(),
222  MAX_EPSILON_ERROR, THRESHOLD, exec_path))
223  {
224  g_TotalErrors++;
225  }
226  }

248 void display()
249 {
250  sdkStartTimer(&timer);
251 
252  // update the simulation
253  if (!bPause)
254  {
255  psystem->setIterations(iterations);
256  psystem->setDamping(damping);
257  psystem->setGravity(-gravity);
258  psystem->setCollideSpring(collideSpring);
259  psystem->setCollideDamping(collideDamping);
260  psystem->setCollideShear(collideShear);
261  psystem->setCollideAttraction(collideAttraction);
262 
263  licznik++;
264  bigRadius=bigRadius0-A*sqrt(licznik*timestep);//r=r0-A*sqrt(t)
265  psystem->setBigRadius(bigRadius);
266 
267  psystem->update(timestep);
268 
269  if (renderer)
270  {
271  renderer->setVertexBuffer(psystem->getCurrentReadBuffer(), psystem->getNumParticles());
272  }
273  }
274 
275  // render
276  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
277 
278  // view transform
279  glMatrixMode(GL_MODELVIEW);
280  glLoadIdentity();
281 
282  for (int c = 0; c < 3; ++c)
283  {
284  camera_trans_lag[c] += (camera_trans[c] - camera_trans_lag[c]) * inertia;
285  camera_rot_lag[c] += (camera_rot[c] - camera_rot_lag[c]) * inertia;
286  }
287 
288  glTranslatef(camera_trans_lag[0], camera_trans_lag[1], camera_trans_lag[2]);
289  glRotatef(camera_rot_lag[0], 1.0, 0.0, 0.0);
290  glRotatef(camera_rot_lag[1], 0.0, 1.0, 0.0);
291 
292  glGetFloatv(GL_MODELVIEW_MATRIX, modelView);
293 
294  // cube -> sphere
295  glColor3f(1.0, 1.0, 1.0);
296  //glutWireCube(2.0);
297  glutWireSphere(bigRadius, 20, 10);
298 
299  // collider
300  glPushMatrix();
301  float3 p = psystem->getColliderPos();
302  glTranslatef(p.x, p.y, p.z);
303  glColor3f(1.0, 0.0, 0.0);
304  glutSolidSphere(psystem->getColliderRadius(), 20, 10);
305  glPopMatrix();
306 
307  if (renderer && displayEnabled)
308  {
309  renderer->display(displayMode);
310  }
311 
312  if (displaySliders)
313  {
314  glDisable(GL_DEPTH_TEST);
315  glBlendFunc(GL_ONE_MINUS_DST_COLOR, GL_ZERO); // invert color
316  glEnable(GL_BLEND);
317  params->Render(0, 0);
318  glDisable(GL_BLEND);
319  glEnable(GL_DEPTH_TEST);
320  }
321 
322  sdkStopTimer(&timer);
323 
324  glutSwapBuffers();
325  glutReportErrors();
326 
327  computeFPS();

335 void addSphere()
336 {
337  // inject a sphere of particles
338  float pr = psystem->getParticleRadius();
339  float tr = pr+(pr*2.0f)*ballr;
340  float pos[4], vel[4];
341  pos[0] = -1.0f + tr + frand()*(2.0f - tr*2.0f);
342  pos[1] = 1.0f - tr;
343  pos[2] = -1.0f + tr + frand()*(2.0f - tr*2.0f);
344  pos[3] = 0.0f;
345  //vel[0] = vel[1] = vel[2] = vel[3] = 0.0f;
346  vel[0]=frand()/100.0f;
347  vel[1]=frand()/100.0f;
348  vel[2]=frand()/100.0f;
349  vel[3]=frand()/100.0f;
350  psystem->addSphere(0, pos, vel, ballr, pr*2.0f);
351 }

411 // transfrom vector by matrix
412 void xform(float *v, float *r, GLfloat *m)
413 {
414  r[0] = v[0]*m[0] + v[1]*m[4] + v[2]*m[8] + m[12];
415  r[1] = v[0]*m[1] + v[1]*m[5] + v[2]*m[9] + m[13];
416  r[2] = v[0]*m[2] + v[1]*m[6] + v[2]*m[10] + m[14];
417 }
418 
419 // transform vector by transpose of matrix
420 void ixform(float *v, float *r, GLfloat *m)
421 {
422  r[0] = v[0]*m[0] + v[1]*m[1] + v[2]*m[2];
423  r[1] = v[0]*m[4] + v[1]*m[5] + v[2]*m[6];
424  r[2] = v[0]*m[8] + v[1]*m[9] + v[2]*m[10];
425 }

427 void ixformPoint(float *v, float *r, GLfloat *m)
428 {
429  float x[4];
430  x[0] = v[0] - m[12];
431  x[1] = v[1] - m[13];
432  x[2] = v[2] - m[14];
433  x[3] = 1.0f;
434  ixform(x, r, m);
435 }

437 void motion(int x, int y)
438 {
439  float dx, dy;
440  dx = (float)(x - ox);
441  dy = (float)(y - oy);
442 
443  if (displaySliders)
444  {
445  if (params->Motion(x, y))
446  {
447  ox = x;
448  oy = y;
449  glutPostRedisplay();
450  return;
451  }
452  }
453 
454  switch (mode)
455  {
456  case M_VIEW:
457  if (buttonState == 3)
458  {
459  // left+middle = zoom
460  camera_trans[2] += (dy / 100.0f) * 0.5f * fabs(camera_trans[2]);
461  }
462  else if (buttonState & 2)
463  {
464  // middle = translate
465  camera_trans[0] += dx / 100.0f;
466  camera_trans[1] -= dy / 100.0f;
467  }
468  else if (buttonState & 1)
469  {
470  // left = rotate
471  camera_rot[0] += dy / 5.0f;
472  camera_rot[1] += dx / 5.0f;
473  }
474 
475  break;
476 
477  case M_MOVE:
478  {
479  float translateSpeed = 0.003f;
480  float3 p = psystem->getColliderPos();
481 
482  if (buttonState==1)
483  {
484  float v[3], r[3];
485  v[0] = dx*translateSpeed;
486  v[1] = -dy*translateSpeed;
487  v[2] = 0.0f;
488  ixform(v, r, modelView);
489  p.x += r[0];
490  p.y += r[1];
491  p.z += r[2];
492  }
493  else if (buttonState==2)
494  {
495  float v[3], r[3];
496  v[0] = 0.0f;
497  v[1] = 0.0f;
498  v[2] = dy*translateSpeed;
499  ixform(v, r, modelView);
500  p.x += r[0];
501  p.y += r[1];
502  p.z += r[2];
503  }
504 
505  psystem->setColliderPos(p);
506  }
507  break;
508  }
509 
510  ox = x;
511  oy = y;
512 
513  demoMode = false;
514  idleCounter = 0;
515 
516  glutPostRedisplay();
517 }
