417 void addSphere()
418 {
419  // inject a sphere of particles
420  float pr = psystem->getParticleRadius();
421  float tr = pr+(pr*2.0f)*ballr;
422  float pos[4], vel[4];
423  pos[0] = -1.0f + tr + frand()*(2.0f - tr*2.0f);
424  pos[1] = 1.0f - tr;
425  pos[2] = -1.0f + tr + frand()*(2.0f - tr*2.0f);
426  pos[3] = 0.0f;
427  //vel[0] = vel[1] = vel[2] = vel[3] = 0.0f;
428  vel[0]=frand()/100.0f;
429  vel[1]=frand()/100.0f;
430  vel[2]=frand()/100.0f;
431  vel[3]=frand()/100.0f;
432  psystem->addSphere(0, pos, vel, ballr, pr*2.0f);

506 // transfrom vector by matrix
507 void xform(float *v, float *r, GLfloat *m)
508 {
509  r[0] = v[0]*m[0] + v[1]*m[4] + v[2]*m[8] + m[12];
510  r[1] = v[0]*m[1] + v[1]*m[5] + v[2]*m[9] + m[13];
511  r[2] = v[0]*m[2] + v[1]*m[6] + v[2]*m[10] + m[14];
512 }

514 // transform vector by transpose of matrix
515 void ixform(float *v, float *r, GLfloat *m)
516 {
517  r[0] = v[0]*m[0] + v[1]*m[1] + v[2]*m[2];
518  r[1] = v[0]*m[4] + v[1]*m[5] + v[2]*m[6];
519  r[2] = v[0]*m[8] + v[1]*m[9] + v[2]*m[10];
520 }

529  ixform(x, r, m);

532 void motion(int x, int y)
533 {
534  float dx, dy;
535  dx = (float)(x - ox);
536  dy = (float)(y - oy);
537 
538  if (displaySliders)
539  {
540  if (params->Motion(x, y))
541  {
542  ox = x;
543  oy = y;
544  glutPostRedisplay();
545  return;
546  }
547  }
548 
549  switch (mode)
550  {
551  case M_VIEW:
552  if (buttonState == 3)
553  {
554  // left+middle = zoom
555  camera_trans[2] += (dy / 100.0f) * 0.5f * fabs(camera_trans[2]);
556  }
557  else if (buttonState & 2)
558  {
559  // middle = translate
560  camera_trans[0] += dx / 100.0f;
561  camera_trans[1] -= dy / 100.0f;
562  }
563  else if (buttonState & 1)
564  {
565  // left = rotate
566  camera_rot[0] += dy / 5.0f;
567  camera_rot[1] += dx / 5.0f;
568  }
569 
570  break;
571 
572  case M_MOVE:
573  {
574  float translateSpeed = 0.003f;
575  float3 p = psystem->getColliderPos();
576 
577  if (buttonState==1)
578  {
579  float v[3], r[3];
580  v[0] = dx*translateSpeed;
581  v[1] = -dy*translateSpeed;
582  v[2] = 0.0f;
583  ixform(v, r, modelView);
584  p.x += r[0];
585  p.y += r[1];
586  p.z += r[2];
587  }
588  else if (buttonState==2)
589  {
590  float v[3], r[3];
591  v[0] = 0.0f;
592  v[1] = 0.0f;
593  v[2] = dy*translateSpeed;
594  ixform(v, r, modelView);
595  p.x += r[0];
596  p.y += r[1];
597  p.z += r[2];
598  }
599 
600  psystem->setColliderPos(p);
601  }
602  break;
603  }
604 
605  ox = x;
606  oy = y;
607 
608  demoMode = false;
609  idleCounter = 0;
610 
611  glutPostRedisplay();
612 }
