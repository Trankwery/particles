54 void ParticleRenderer::_drawPoints()
55 {
56  if (!m_vbo)
57  {
58  glBegin(GL_POINTS);
59  {
60  int k = 0;
61 
62  for (int i = 0; i < m_numParticles; ++i)
63  {
64  glVertex3fv(&m_pos[k]);
65  k += 4;
66  }
67  }
68  glEnd();
69  }
70  else
71  {
72  glBindBufferARB(GL_ARRAY_BUFFER_ARB, m_vbo);
73  glVertexPointer(4, GL_FLOAT, 0, 0);
74  glEnableClientState(GL_VERTEX_ARRAY);
75 
76  if (m_colorVBO)
77  {
78  glBindBufferARB(GL_ARRAY_BUFFER_ARB, m_colorVBO);
79  glColorPointer(4, GL_FLOAT, 0, 0);
80  glEnableClientState(GL_COLOR_ARRAY);
81  }
82 
83  glDrawArrays(GL_POINTS, 0, m_numParticles);
84 
85  glBindBufferARB(GL_ARRAY_BUFFER_ARB, 0);
86  glDisableClientState(GL_VERTEX_ARRAY);
87  glDisableClientState(GL_COLOR_ARRAY);
88  }

91 void ParticleRenderer::display(DisplayMode mode /* = PARTICLE_POINTS */)
92 {
93  switch (mode)
94  {
95  case PARTICLE_POINTS:
96  glColor3f(1, 1, 1);
97  glPointSize(m_pointSize);
98  _drawPoints();
99  break;
100 
101  default:
102  case PARTICLE_SPHERES:
103  glEnable(GL_POINT_SPRITE_ARB);
104  glTexEnvi(GL_POINT_SPRITE_ARB, GL_COORD_REPLACE_ARB, GL_TRUE);
105  glEnable(GL_VERTEX_PROGRAM_POINT_SIZE_NV);
106  glDepthMask(GL_TRUE);
107  glEnable(GL_DEPTH_TEST);
108 
109  glUseProgram(m_program);
110  glUniform1f(glGetUniformLocation(m_program, "pointScale"), m_window_h / tanf(m_fov*0.5f*(float)M_PI/180.0f));
111  glUniform1f(glGetUniformLocation(m_program, "pointRadius"), m_particleRadius);
112 
113  glColor3f(1, 1, 1);
114  _drawPoints();
115 
116  glUseProgram(0);
117  glDisable(GL_POINT_SPRITE_ARB);
118  break;
119  }

123 ParticleRenderer::_compileProgram(const char *vsource, const char *fsource)
124 {
125  GLuint vertexShader = glCreateShader(GL_VERTEX_SHADER);
126  GLuint fragmentShader = glCreateShader(GL_FRAGMENT_SHADER);
127 
128  glShaderSource(vertexShader, 1, &vsource, 0);
129  glShaderSource(fragmentShader, 1, &fsource, 0);
130 
131  glCompileShader(vertexShader);
132  glCompileShader(fragmentShader);
133 
134  GLuint program = glCreateProgram();
135 
136  glAttachShader(program, vertexShader);
137  glAttachShader(program, fragmentShader);
138 
139  glLinkProgram(program);
140 
141  // check if program linked
142  GLint success = 0;
143  glGetProgramiv(program, GL_LINK_STATUS, &success);
144 
145  if (!success)
