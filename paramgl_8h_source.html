30 inline void beginWinCoords(void)
31 {
32  glMatrixMode(GL_MODELVIEW);
33  glPushMatrix();
34  glLoadIdentity();
35  glTranslatef(0.0, (GLfloat)(glutGet(GLUT_WINDOW_HEIGHT) - 1.0), 0.0);
36  glScalef(1.0, -1.0, 1.0);
37 
38  glMatrixMode(GL_PROJECTION);
39  glPushMatrix();
40  glLoadIdentity();
41  glOrtho(0, glutGet(GLUT_WINDOW_WIDTH), 0, glutGet(GLUT_WINDOW_HEIGHT), -1, 1);
42 
43  glMatrixMode(GL_MODELVIEW);
44 }

100  void Render(int x, int y, bool shadow = false)
101  {
102  beginWinCoords();
103 
104  m_start_x = x;
105  m_start_y = y;
106 
107  for (std::vector<ParamBase *>::const_iterator p = m_params.begin(); p != m_params.end(); ++p)
108  {
109  if ((*p)->IsList())
110  {
111  ParamListGL *list = (ParamListGL *)(*p);
112  list->Render(x+10, y);
113  y += m_separation*list->GetSize();
114  }
115  else
116  {
117  if (p == m_current)
118  {
119  glColor3fv(&m_text_color_selected.r);
120  }
121  else
122  {
123  glColor3fv(&m_text_color_unselected.r);
124  }
125 
126  if (shadow)
127  {
128  glPrintShadowed(x + m_text_x, y + m_font_h, (*p)->GetName().c_str(), m_font, (p == m_current) ? &m_text_color_selected.r : &m_text_color_unselected.r);
129  glPrintShadowed(x + m_value_x, y + m_font_h, (*p)->GetValueString().c_str(), m_font, (p == m_current) ? &m_text_color_selected.r : &m_text_color_unselected.r);
130  }
131  else
132  {
133  glPrint(x + m_text_x, y + m_font_h, (*p)->GetName().c_str(), m_font);
134  glPrint(x + m_value_x, y + m_font_h, (*p)->GetValueString().c_str(), m_font);
135  }
136 
137  glColor3fv((GLfloat *) &m_bar_color_outer.r);
138  glBegin(GL_LINE_LOOP);
139  glVertex2f((GLfloat)(x + m_bar_x) , (GLfloat)(y + m_bar_offset));
140  glVertex2f((GLfloat)(x + m_bar_x + m_bar_w), (GLfloat)(y + m_bar_offset));
141  glVertex2f((GLfloat)(x + m_bar_x + m_bar_w), (GLfloat)(y + m_bar_offset + m_bar_h));
142  glVertex2f((GLfloat)(x + m_bar_x) , (GLfloat)(y + m_bar_offset + m_bar_h));
143  glEnd();
144 
145  glColor3fv((GLfloat *) &m_bar_color_inner.r);
146  glRectf((GLfloat)(x + m_bar_x), (GLfloat)(y + m_bar_offset + m_bar_h), (GLfloat)(x + m_bar_x + ((m_bar_w-1)*(*p)->GetPercentage())), (GLfloat)(y + m_bar_offset + 1));
147 
148  y += m_separation;
149  }
150  }
151 
152  endWinCoords();
153  }

155  bool Mouse(int x, int y, int button=GLUT_LEFT_BUTTON, int state=GLUT_DOWN)
156  {
157  if ((y < m_start_y) || (y > (int)(m_start_y + (m_separation * m_params.size()) - 1)))
158  {
159  m_active = false;
160  return false;
161  }
162 
163  m_active = true;
164 
165  int i = (y - m_start_y) / m_separation;
166 
167  if ((button==GLUT_LEFT_BUTTON) && (state==GLUT_DOWN))
168  {
169 #if defined(__GNUC__) && (__GNUC__ < 3)
170  m_current = &m_params[i];
171 #else
172 
173  // MJH: workaround since the version of vector::at used here is non-standard
174  for (m_current = m_params.begin(); m_current != m_params.end() && i > 0; m_current++, i--) ;
175 
176  //m_current = (std::vector<ParamBase *>::const_iterator)&m_params.at(i);
177 #endif
178 
179  if ((x > m_bar_x) && (x < m_bar_x + m_bar_w))
180  {
181  Motion(x, y);
182  }
183  }
184 
185  return true;
186  }

188  bool Motion(int x, int y)
189  {
190  if ((y < m_start_y) || (y > m_start_y + (m_separation * (int)m_params.size()) - 1))
191  {
192  return false;
193  }
194 
195  if (x < m_bar_x)
196  {
197  (*m_current)->SetPercentage(0.0);
198  return true;

200

202  {
203  (*m_current)->SetPercentage(1.0);
204  return true;
205  }
206 
207  (*m_current)->SetPercentage((x-m_bar_x) / (float) m_bar_w);
208  return true;
