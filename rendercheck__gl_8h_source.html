576  virtual bool readback(GLuint width, GLuint height)
577  {
578  bool ret = false;
579 
580  if (m_bUsePBO)
581  {
582  // binds the PBO for readback
583  bindReadback();
584 
585  // Initiate the readback BLT from BackBuffer->PBO->membuf
586  glReadPixels(0, 0, width, height, getPixelFormat(), GL_UNSIGNED_BYTE, BUFFER_OFFSET(0));
587  ret = checkStatus(__FILE__, __LINE__, true);
588 
589  if (!ret)
590  {
591  printf("CheckBackBuffer::glReadPixels() checkStatus = %d\n", ret);
592  }
593 
594  // map - unmap simulates readback without the copy
595  void *ioMem = glMapBufferARB(GL_PIXEL_PACK_BUFFER_ARB, GL_READ_ONLY_ARB);
596  memcpy(m_pImageData, ioMem, width*height*m_Bpp);
597 
598  glUnmapBufferARB(GL_PIXEL_PACK_BUFFER_ARB);
599 
600  // release the PBO
601  unbindReadback();
602  }
603  else
604  {
605  // reading direct from the backbuffer
606  glReadBuffer(GL_FRONT);
607  glReadPixels(0, 0, width, height, getPixelFormat(), GL_UNSIGNED_BYTE, m_pImageData);
608  }
609 
610  return ret;
611  }

613  virtual bool readback(GLuint width, GLuint height, GLuint bufObject)
614  {
615  bool ret = false;
616 
617  if (m_bUseFBO)
618  {
619  if (m_bUsePBO)
620  {
621  printf("CheckBackBuffer::readback() FBO->PBO->m_pImageData\n");
622  // binds the PBO for readback
623  bindReadback();
624 
625  // bind FBO buffer (we want to transfer FBO -> PBO)
626  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, bufObject);
627 
628  // Now initiate the readback to PBO
629  glReadPixels(0, 0, width, height, getPixelFormat(), GL_UNSIGNED_BYTE, BUFFER_OFFSET(0));
630  ret = checkStatus(__FILE__, __LINE__, true);
631 
632  if (!ret)
633  {
634  printf("CheckBackBuffer::readback() FBO->PBO checkStatus = %d\n", ret);
635  }
636 
637  // map - unmap simulates readback without the copy
638  void *ioMem = glMapBufferARB(GL_PIXEL_PACK_BUFFER_ARB, GL_READ_ONLY_ARB);
639  memcpy(m_pImageData, ioMem, width*height*m_Bpp);
640 
641  glUnmapBufferARB(GL_PIXEL_PACK_BUFFER_ARB);
642 
643  // release the FBO
644  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, 0);
645 
646  // release the PBO
647  unbindReadback();
648  }
649  else
650  {
651  printf("CheckBackBuffer::readback() FBO->m_pImageData\n");
652  // Reading direct to FBO using glReadPixels
653  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, bufObject);
654  ret = checkStatus(__FILE__, __LINE__, true);
655 
656  if (!ret)
657  {
658  printf("CheckBackBuffer::readback::glBindFramebufferEXT() fbo=%d checkStatus = %d\n", bufObject, ret);
659  }
660 
661  glReadBuffer(static_cast<GLenum>(GL_COLOR_ATTACHMENT0_EXT));
662  ret &= checkStatus(__FILE__, __LINE__, true);
663 
664  if (!ret)
665  {
666  printf("CheckBackBuffer::readback::glReadBuffer() fbo=%d checkStatus = %d\n", bufObject, ret);
667  }
668 
669  glReadPixels(0, 0, width, height, getPixelFormat(), GL_UNSIGNED_BYTE, m_pImageData);
670 
671  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, 0);
672  }
673  }
674  else
675  {
676 
677  printf("CheckBackBuffer::readback() PBO->m_pImageData\n");
678  // read from bufObject (PBO) to system memorys image
679  glBindBufferARB(GL_PIXEL_PACK_BUFFER_ARB, bufObject); // Bind the PBO
680 
681  // map - unmap simulates readback without the copy
682  void *ioMem = glMapBufferARB(GL_PIXEL_PACK_BUFFER_ARB, GL_READ_ONLY_ARB);
683 
684  // allocate a buffer so we can flip the image
685  unsigned char *temp_buf = (unsigned char *)malloc(width*height*m_Bpp);
686  memcpy(temp_buf, ioMem, width*height*m_Bpp);
687 
688  // let's flip the image as we copy
689  for (unsigned int y = 0; y < height; y++)
690  {
691  memcpy((void *)&(m_pImageData[(height-y)*width*m_Bpp]), (void *)&(temp_buf[y*width*m_Bpp]), width*m_Bpp);
692  }
693 
694  free(temp_buf);
695 
696  glUnmapBufferARB(GL_PIXEL_PACK_BUFFER_ARB);
697 
698  // read from bufObject (PBO) to system memory image
699  glBindBufferARB(GL_PIXEL_PACK_BUFFER_ARB, 0); // unBind the PBO
700  }
701 
702  return CHECK_FBO;
703  }

892  bool create(GLuint width, GLuint height, fboConfig &config, fboData &data)
893  {
894  bool multisample = config.depthSamples > 0;
895  bool ret = true;
896  GLint query;
897 
898  printf("\nCreating FBO <%s> (%dx%d) Float:%s\n", config.name.c_str(), (int)width, (int)height, (m_bUseFloat ? "Y":"N"));
899

906  width, height, 0, GL_RGBA,

908  NULL);

911

915  glTexParameterf(m_eGLTarget, GL_TEXTURE_MAG_FILTER, GL_NEAREST); // GL_LINEAR);
916 
917  {
918  glGenTextures(1, &data.depthTex);
919  data.depthRB = 0;
920  data.colorRB = 0;
921  data.resolveFB = 0;

924  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, data.fb);
925  glFramebufferTexture2DEXT(GL_FRAMEBUFFER_EXT, GL_COLOR_ATTACHMENT0_EXT, m_eGLTarget, data.colorTex, 0);
926

928  glTexImage2D(m_eGLTarget, 0, config.depthFormat,
929  width, height, 0, GL_DEPTH_COMPONENT, GL_FLOAT, NULL);
930

932  glTexParameterf(m_eGLTarget, GL_TEXTURE_MAG_FILTER, GL_NEAREST); // GL_LINEAR);

934  glTexParameterf(m_eGLTarget, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
935  glTexParameterf(m_eGLTarget, GL_DEPTH_TEXTURE_MODE, GL_LUMINANCE);
936

938 
939  ret &= checkStatus(__FILE__, __LINE__, true);
940  }

944

946  {
947  ret = false;
948  }
949 
950  glGetIntegerv(GL_DEPTH_BITS, &query);
951

953  {
954  ret = false;
955  }
956 
957  if (multisample)
958  {
959  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, data.resolveFB);
960  glGetIntegerv(GL_RED_BITS, &query);
961 
962  if (query != config.redbits)
963  {
964  ret = false;
965  }
966  }
967 
968  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, 0);
969 
970  ret &= checkStatus(__FILE__, __LINE__, true);
971
