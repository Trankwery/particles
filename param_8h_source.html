59 // derived class for single-valued parameter
60 template<class T> class Param : public ParamBase
61 {
62  public:
63  Param(const char *name, T value = 0, T min = 0, T max = 10000, T step = 1, T *ptr = 0) :
64  ParamBase(name),
65  m_default(value),
66  m_min(min),
67  m_max(max),
68  m_step(step),
69  m_precision(5)
70  {
71  if(max<value)
72  {
73  m_max=3*value;
74  }
75  if (ptr)
76  {
77  m_ptr = ptr;
78  }
79  else
80  {
81  m_ptr = &m_value;
82  }
83 
84  *m_ptr = value;
85  }
86  ~Param() { }
87

89  {
90  return *m_ptr;
91  }

94  *m_ptr = value;
95  }

97  float GetFloatValue()

99  return (float) *m_ptr;
100  }
101  int GetIntValue()
102  {
103  return (int) *m_ptr;
104  }
105 
106  std::string GetValueString()
107  {
108  std::ostringstream ost;
109  ost<<std::setprecision(m_precision)<<std::fixed;
110  ost<<*m_ptr;

112  }

114  void SetPrecision(int x)
115  {
116  m_precision = x;

118 
119  float GetPercentage()
120  {
121  return (*m_ptr - m_min) / (float)(m_max - m_min);
122  }
123 
124  void SetPercentage(float p)
125  {
126  *m_ptr = (T)(m_min + p * (m_max - m_min));
127  }
128 
129  void Reset()
130  {
131  *m_ptr = m_default;
132  }
133 
134  void Increment()
135  {
136  *m_ptr += m_step;
137 
138  if (*m_ptr > m_max)
139  {
140  *m_ptr = m_max;
141  }
142  }
143 
144  void Decrement()
145  {
146  *m_ptr -= m_step;
147 
148  if (*m_ptr < m_min)
149  {
150  *m_ptr = m_min;
151  }
152  }
153 
154  void Write(std::ostream &stream)
155  {
156  stream << m_name << " " << *m_ptr << '\n';
157  }
158  void Read(std::istream &stream)
159  {
160  stream >> m_name >> *m_ptr;
161  }
162 
163  bool IsList()
164  {
165  return false;
166  }

168  private:

172  int m_precision; // number of digits after decimal point in string output
173 };
